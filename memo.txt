apk add --no-cache git protobuf curl
go get -u github.com/grpc-ecosystem/grpc-gateway/protoc-gen-grpc-gateway 
go get -u github.com/golang/protobuf/protoc-gen-go

cd /tmp

curl -OL https://github.com/google/protobuf/releases/download/v3.0.0/protoc-3.0.0-linux-x86_64.zip
unzip protoc-3.0.0-linux-x86_64.zip
mv include /usr/local/.

rm -f readme.txt protoc-3.0.0-linux-x86_64.zip

cd $GOPATH/src/

mkdir -p authenticationservice/src/genproto

cd authenticationservice

touch authentication.proto

ls -l $(find $(pwd) | tail -1) // => -rw-r--r--    1 root     root           349 Sep  4 04:35 /go/src/authenticationservice/pd/authentication.proto
cd ../

export PATH=$PATH:$GOPATH/bin

protoc -I/usr/local/include -I. \
  -I$GOPATH/src \
  -I$GOPATH/src/github.com/grpc-ecosystem/grpc-gateway/third_party/googleapis \
  --go_out=plugins=grpc:./src/genproto \
  authentication.proto

protoc -I/usr/local/include -I. \
  -I$GOPATH/src \
  -I$GOPATH/src/github.com/grpc-ecosystem/grpc-gateway/third_party/googleapis \
  --js_out=. \
  authentication.proto

protoc -I/usr/local/include -I. \
  -I$GOPATH/src \
  -I$GOPATH/src/github.com/grpc-ecosystem/grpc-gateway/third_party/googleapis \
  --grpc-gateway_out=logtostderr=true:. \
  authentication.proto
